#+end_src

* Create di = ls ^ cat alias
#+begin_src fish
function ccat
    if type -q bat
        bat $argv
    else
        cat $argv
    end
end
function di
    argparse --name=di -i 'l' 'c' -- $argv
    set _dir $argv[1]
    if test "$_flag_c" = "-c"
        ccat $argv
    else if test "$_flag_l" = "-l"; or test -d $_dir
        ls -hAl $argv
    else
        ccat $argv
    end
end
#+end_src

* Alias to untar
#+begin_src fish

#+end_src

* Start starship
#+begin_src fish

        source "$HOME/.config/fish/starship_cygwin.fish"

        starship init fish | source
